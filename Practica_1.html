Paso 7: Evaluación del modelo¶

# Realizar predicciones en el conjunto de prueba
y_pred = model.predict(X_test)

# Calcular el error cuadrtico medio (MSE)
mse = mean_squared_error(y_test, y_pred)
print('Error cuadrtico medio (MSE):', mse)

# Calcular el coeficiente de determinacin (R2-score)
r2 = r2_score(y_test, y_pred)
print('Coeficiente de determinacin (R2-score):', r2)

print("El R cuadrado del modelo es: "+str(model.score(X_train,y_train)))

print("Los coeficientes de la funcion son: "+str(model.coef_))
rmse =np.sqrt(mean_squared_error(y_test,y_pred))
print("Root Mean Squared error: {}".format(rmse))
